* Variables
Some variables with path to plugins and items directories. Also with spotify event name (Not using it right now).
#+begin_src shell
PLUGIN_DIR="$HOME/.config/sketchybar/horizontal/plugins"
ITEM_DIR="$HOME/.config/sketchybar/horizontal/items"
SPOTIFY_EVENT="com.spotify.client.PlaybackStateChanged"
#+end_src

* Source

  Sourcing some colors and icons

#+begin_src shell
source $HOME/.config/sketchybar/colors.sh
source $HOME/.config/sketchybar/icons.sh
#+end_src

* Exports
#+begin_src shell
export HORIZOTAL_PADDING=10
export LABEL_HORIZONTAL_PADDING=4
export HEIGHT=38
export FONT="JetBrainsMono Nerd Font Mono"
export NERD_FONT="JetBrainsMono Nerd Font Mono"
#+end_src

* Bar general settings

#+begin_src shell
bar=(
	position=top
	height=$HEIGHT
	padding_left=$HORIZOTAL_PADDING
	padding_right=$HORIZOTAL_PADDING
	color=$BACKGROUND
	shadow=off
	sticky=on
)
sketchybar	--bar	"${bar[@]}"
#+end_src

* Default settings

#+begin_src shell
default=(
	icon.font="$FONT:Bold:15.0"  
	icon.color=$FOREGROUND       
	label.font="$FONT:Bold:15.0" 
	label.color=$FOREGROUND      
	label.padding_left=$LABEL_HORIZONTAL_PADDING
	label.padding_right=$LABEL_HORIZONTAL_PADDING
	popup.background.color=$BACKGROUND
	popup.background.corner_radius=5
	popup.background.border_width=2
	popup.background.border_color=$BLUE
	popup.background.shadow.drawing=on
)

sketchybar	--default	"${default[@]}" 
#+end_src
